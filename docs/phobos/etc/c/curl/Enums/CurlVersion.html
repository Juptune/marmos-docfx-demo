enum CurlVersion
{
  // asynchronous dns resolves 
  __enumMember asynchdns;
  // character conversions supported 
  __enumMember conv;
  // debug memory tracking supported 
  __enumMember curldebug;
  // built with debug capabilities 
  __enumMember dbg;
  // Negotiate auth support 
  __enumMember gssnegotiate;
  // International Domain Names support 
  __enumMember idn;
  // IPv6-enabled 
  __enumMember ipv6;
  // kerberos auth is supported 
  __enumMember kerberos4;
  // supports files bigger than 2GB 
  __enumMember largefile;
  // libz features are present 
  __enumMember libz;
  // NTLM auth is supported 
  __enumMember ntlm;
  // SPNEGO auth 
  __enumMember spnego;
  // SSL options are present 
  __enumMember ssl;
  // SSPI is supported 
  __enumMember sspi;
  // TLS-SRP auth is supported 
  __enumMember tlsauth_srp;
}
