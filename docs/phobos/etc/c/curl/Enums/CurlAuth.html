Enum - CurlAuth

Module
etc.c.curl
enum CurlAuth
{
  // any allows 
  __enumMember any;
  // any except basic 
  __enumMember anysafe;
  // Basic (default) 
  __enumMember basic;
  // Bearer token authentication 
  __enumMember bearer;
  // Digest 
  __enumMember digest;
  // Digest with IE flavour 
  __enumMember digest_ie;
  // GSS-Negoatiate 
  __enumMember gssapi;
  // GSS-Negotiate 
  __enumMember gssnegotiate;
  // Negotiate (SPNEGO) 
  __enumMember negotiate;
  // None 
  __enumMember none;
  // NTLM 
  __enumMember ntlm;
  // NTML delegated to winbind helper 
  __enumMember ntlm_WB;
  // used together with a single other type to force no auth or just that single type 
  __enumMember only;
}
