enum SocketFlags
{
  // data should not be subject to routing; this flag may be ignored. Only for sending 
  __enumMember DONTROUTE;
  // no flags specified 
  __enumMember NONE;
  // out-of-band stream data 
  __enumMember OOB;
  // peek at incoming data without removing it from the queue, only for receiving 
  __enumMember PEEK;
}
