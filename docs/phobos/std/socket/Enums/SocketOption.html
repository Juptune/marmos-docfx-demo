Enum - SocketOption

Module
std.socket
enum SocketOption
{
  // Listen 
  __enumMember ACCEPTCONN;
  // Allow transmission of broadcast messages 
  __enumMember BROADCAST;
  // Record debugging information 
  __enumMember DEBUG;
  // Do not route 
  __enumMember DONTROUTE;
  // Retrieve and clear error status 
  __enumMember ERROR;
  // Add an IP group membership 
  __enumMember IPV6_JOIN_GROUP;
  // Drop an IP group membership 
  __enumMember IPV6_LEAVE_GROUP;
  // IP multicast hops 
  __enumMember IPV6_MULTICAST_HOPS;
  // IP multicast interface 
  __enumMember IPV6_MULTICAST_IF;
  // IP multicast loopback 
  __enumMember IPV6_MULTICAST_LOOP;
  // IP unicast hop limit 
  __enumMember IPV6_UNICAST_HOPS;
  // Treat wildcard bind as AF_INET6-only 
  __enumMember IPV6_V6ONLY;
  // Enable keep-alive packets 
  __enumMember KEEPALIVE;
  // Linger on close if unsent data is present 
  __enumMember LINGER;
  // Receive out-of-band data in band 
  __enumMember OOBINLINE;
  // Receive buffer size 
  __enumMember RCVBUF;
  // Minimum number of input bytes to process 
  __enumMember RCVLOWAT;
  // Receive timeout 
  __enumMember RCVTIMEO;
  // Allow local reuse of address 
  __enumMember REUSEADDR;
  // Send buffer size 
  __enumMember SNDBUF;
  // Minimum number of output bytes to process 
  __enumMember SNDLOWAT;
  // Send timeout 
  __enumMember SNDTIMEO;
  // Disable the Nagle algorithm for send coalescing 
  __enumMember TCP_NODELAY;
  // Socket type 
  __enumMember TYPE;
}
