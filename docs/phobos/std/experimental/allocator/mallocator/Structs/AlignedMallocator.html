  bool deallocate( <bug: unknown> b ) ;
  // Forwards to $(D alignedReallocate(b, newSize, platformAlignment)). Should be used with blocks obtained with `allocate` otherwise the custom alignment passed with `alignedAllocate` can be lost.

About
